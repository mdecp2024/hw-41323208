def print_filled_circle(radius, center_x, center_y, size=10):
    for y in range(size):
        for x in range(size):
            # 計算 (x, y) 點到圓心的距離
            distance = (x - center_x) ** 2 + (y - center_y) ** 2
            # 如果距離小於或等於半徑的平方，則是圓內的點，打印 *
            if distance <= radius ** 2:
                print('*', end='')
            else:
                print(' ', end='')
        print()

# 設定圓心在 (4, 4)，半徑為 4
print_filled_circle(radius=4, center_x=4, center_y=4)

from browser import html
from browser import document as doc

# 创建画布
canvas = html.CANVAS(width=400, height=400)
brython_div = doc["brython_div1"]
brython_div <= canvas

# 每个方块的大小
gs = 20

# 获取画布的 2D 上下文
ctx = canvas.getContext("2d")

# 绘制矩形的函数，带黑色填充
def dRect(lux, luy, w, h, c='#ff0000'):
    ctx.fillStyle = c  # 填充颜色
    ctx.fillRect(lux, luy, w, h)  # 绘制填充矩形

# 绘制由黑色方块拼成的圆形图案
def drawCircleWithFilledBlackSquares(radius):
    # 画布的中心点
    cx, cy = canvas.width // 2, canvas.height // 2
    # 计算网格数
    for i in range(-radius, radius, gs):
        for j in range(-radius, radius, gs):
            # 当前网格的中心坐标
            x = cx + i
            y = cy + j
            # 计算该网格到中心的距离
            dist = (i ** 2 + j ** 2) ** 0.5
            # 如果该网格的距离小于圆的半径，填充方块
            if dist < radius:
                dRect(x, y, gs, gs, "black")  # 填充黑色

# 绘制由黑色方块填充的圆形
drawCircleWithFilledBlackSquares(100)
