以下為 12 號的程式:

import math

def draw_circle_10x10():
    size = 10  # 固定為 10x10 的字元區域
    radius = 4.5  # 使用半徑 4.5 來製作圓形，優先填滿 10x10 區域
    center = (size - 1) / 2  # 圓心位置在 10x10 區域的正中央

    # 建立 10x10 的字元矩陣
    for y in range(size):
        for x in range(size):
            # 計算 (x, y) 到圓心的距離
            distance = math.sqrt((x - center) ** 2 + (y - center) ** 2)
            # 使用範圍判斷來決定是否填入 '*'
            if radius - 0.5 <= distance <= radius + 0.5:
                print('*', end='')
            else:
                print(' ', end='')
        print()  # 列印每一行後換行

# 執行 10x10 圓形繪製
draw_circle_10x10()
